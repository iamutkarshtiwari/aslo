Verify_simple_login_to AMO_firefox
store amo.testing@gmail.com email
store 1234 password
open /en-US/firefox/
storeElementPresent link=Log in value
echo Is the tester logged in: ${value}
gotoIf "${value}"=="true" target
clickAndWait link=Log out
label target
clickAndWait link=Log in
type LoginEmail ${email}
type LoginPassword ${password}
check LoginRemember
clickAndWait css=#login button[type="submit"]
verifyText css=a[href="/en-US/firefox/users/edit"] My Account
verifyText css=a[href="/en-US/firefox/users/logout?to=en-US%2Ffirefox%2F"] Log out
clickAndWait link=Log out
verifyLocation regexpi:.*addons.mozilla.org/en-US/firefox/
echo DONE
